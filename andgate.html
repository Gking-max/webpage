// Function to toggle the first input image and state let inputOne = false; // Track the state of input 1 function toggleImage() { const img = document.getElementById("toggleImage"); inputOne = !inputOne; // Toggle the state if (inputOne) { img.src = "https://www.iconsdb.com/icons/preview/green/button-on-xxl.png"; } else { img.src = "https://www.iconsdb.com/icons/preview/red/button-off-xxl.png"; } } // Function to toggle the second input image and state let inputTwo = false; // Track the state of input 2 function toggleImage2() { const img = document.getElementById("toggleImage2"); inputTwo = !inputTwo; // Toggle the state if (inputTwo) { img.src = "https://www.iconsdb.com/icons/preview/green/button-on-xxl.png"; } else { img.src = "https://www.iconsdb.com/icons/preview/red/button-off-xxl.png"; } } // AND gate function to set the output image based on both inputs function and() { const andGate = document.getElementById("andGate"); if (inputOne && inputTwo) { andGate.src = "and2on.PNG"; // Display the "on" AND gate image when both inputs are on } else { andGate.src = "and2off.PNG"; // Display the "off" AND gate image when either input is off } }
